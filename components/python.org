#+TITLE: python
#+PROPERTY: header-args :tangle yes

Python development using the built-in =python.el= and =eglot= for completion,
navigation and diagnostics.  Supported language servers (first found on PATH
wins): =basedpyright= (recommended — community-maintained Pyright fork),
=pyright=, =pylsp=, or =jedi-language-server=.  Install via pip or Homebrew.
For formatting, Apheleia uses =ruff= automatically when the =python= component
is active alongside the =apheleia= component.

#+BEGIN_SRC emacs-lisp
(setq python-shell-interpreter "python3"
      python-shell-interpreter-args "-i")

(defun kwarks/python-setup ()
  (setq-local python-indent-offset 4
              tab-width 4)
  (kwarks/eglot-ensure-if "basedpyright-langserver" "pyright-langserver" "pyright"
                          "pylsp" "jedi-language-server"))

(add-hook 'python-mode-hook #'kwarks/python-setup)
(add-hook 'python-ts-mode-hook #'kwarks/python-setup)

(when (and (fboundp 'treesit-ready-p) (treesit-ready-p 'python t))
  (add-to-list 'major-mode-remap-alist '(python-mode . python-ts-mode)))
#+END_SRC
